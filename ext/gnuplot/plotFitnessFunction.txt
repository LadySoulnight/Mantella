#!/usr/bin/gnuplot

# Reset previous configuration
reset

# Loading new default color palette
load 'config.txt'
set terminal pngcairo size 1024,768 enhanced font 'Verdana, 10'

set output 'fitnessFunction.png'

set hidden3d

set surface
set contour

set grid xtics
set grid ytics
set grid ztics
set grid nomxtics
set grid nomytics
set grid nomztics

set style line 1 linecolor rgb '#808080' linetype 1 linewidth 0.5

# axes
set style line 11 linecolor rgb '#808080' lt 1
set border 3 front linestyle 11

# 
set tics nomirror
set tics out
set tics scale 0.75

# grid
set style line 12 linecolor rgb'#808080' linetype 0 linewidth 1
set grid back linestyle 12

set xrange [-5:5]
set yrange [-5:5]

set xlabel 'Parameter (x_1)'
set ylabel 'Parameter (x_2)'
set zlabel 'Fitness value (f)' rotate center


unset clabel
set cntrparam levels 20

splot 'fitnessFunction.dat' with pm3d palette
